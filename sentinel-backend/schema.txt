// prisma/schema.prisma
// Prisma Schema for Patroller — Police Patrolling Verification System

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ─────────────────────────────────────────────
// Enums
// ─────────────────────────────────────────────

enum EventType {
  NAKA
  VIP_SECURITY
  CROWD_CONTROL
  BANDOBAST
  PATROL
  OTHER
}

enum Shift {
  MORNING
  AFTERNOON
  NIGHT
}

enum DutyStatus {
  PENDING
  ACTIVE
  COMPLETED
  ABSENT
}

enum Severity {
  LOW
  MEDIUM
  HIGH
  CRITICAL
}

// ─────────────────────────────────────────────
// Models
// ─────────────────────────────────────────────

model Admin {
  id        String   @id @default(cuid())
  name      String
  email     String   @unique
  phone     String
  password  String
  imageURL  String?
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  stations  Station[]
  events    Event[]

  @@map("admins")
}

model Station {
  id       String @id @default(cuid())
  name     String
  location String
  code     String @unique
  adminID  String

  admin      Admin       @relation(fields: [adminID], references: [id])
  patrollers Patroller[]

  @@map("stations")
}

model Patroller {
  id        String  @id @default(cuid())
  name      String
  email     String  @unique
  phone     String
  password  String
  stationID String
  post      String
  imageURL  String?

  station      Station       @relation(fields: [stationID], references: [id])
  assignments  Assigned[]
  deviceTokens DeviceToken[]

  @@map("patrollers")
}

model Event {
  id        String    @id @default(cuid())
  name      String
  eventType EventType
  startDate DateTime  @db.Date
  endDate   DateTime  @db.Date
  location  String
  adminID   String

  admin  Admin   @relation(fields: [adminID], references: [id])
  duties Duty[]

  @@map("events")
}

model Duty {
  id        String   @id @default(cuid())
  name      String
  shift     Shift
  zone      String
  date      DateTime @db.Date
  startTime DateTime @db.Time(0)
  endTime   DateTime @db.Time(0)
  eventName String
  eventID   String

  event       Event      @relation(fields: [eventID], references: [id])
  assignments Assigned[]

  @@map("duties")
}

model Assigned {
  id          String @id @default(cuid())
  patrollerID String
  dutyID      String

  patroller   Patroller    @relation(fields: [patrollerID], references: [id])
  duty        Duty         @relation(fields: [dutyID], references: [id])
  assignedLog AssignedLog?
  incidents   Incident[]
  telemetry   Telemetry[]

  @@unique([patrollerID, dutyID])
  @@map("assigned")
}

model AssignedLog {
  id           String     @id @default(cuid())
  dutyStatus   DutyStatus
  faceVerified Boolean    @default(false)
  checkinTime  DateTime?  @db.Time(0)
  deletedAt    DateTime?
  patrollerID  String
  dutyID       String
  assignedID   String     @unique

  assigned Assigned @relation(fields: [assignedID], references: [id])

  @@map("assigned_logs")
}

model Incident {
  id          String   @id @default(cuid())
  name        String
  severity    Severity
  timeStamp   DateTime
  imageUrl    String?
  description String
  assignedID  String

  assigned Assigned @relation(fields: [assignedID], references: [id])

  @@map("incidents")
}

model Telemetry {
  id         String   @id @default(cuid())
  timeStamp  DateTime @default(now())
  withinZone Boolean
  lat        String
  lng        String
  assignedID String

  assigned Assigned @relation(fields: [assignedID], references: [id])

  @@map("telemetry")
}

model DeviceToken {
  id          String   @id @default(cuid())
  token       String   @unique
  createdAt   DateTime @default(now())
  patrollerID String

  patroller Patroller @relation(fields: [patrollerID], references: [id])

  @@map("device_tokens")
}
